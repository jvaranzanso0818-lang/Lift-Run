<script>
  const STORAGE_KEY = 'forge_fitness_data';
  let data = {};
  let viewYear, viewMonth;
  let selectedDate = null;

  // --- Data ---
  function loadData() {
    try { data = JSON.parse(localStorage.getItem(STORAGE_KEY)) || {}; }
    catch { data = {}; }
  }

  function saveData() {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
  }

  // --- Date helpers ---
  function toKey(y, m, d) {
    return `${y}-${String(m+1).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
  }

  function todayKey() {
    const t = new Date();
    return toKey(t.getFullYear(), t.getMonth(), t.getDate());
  }

  function addDays(dateStr, n) {
    const d = new Date(dateStr + 'T00:00:00');
    d.setDate(d.getDate() + n);
    return d.toISOString().slice(0,10);
  }

  // --- Streak ---
  function calcStreak() {
    let streak = 0;
    let cur = todayKey();
    while (true) {
      const t = data[cur];
      if (t === 'lift' || t === 'run') {
        streak++;
        cur = addDays(cur, -1);
      } else { break; }
    }
    return streak;
  }

  function renderStreak() {
    document.getElementById('streakVal').textContent = calcStreak();
  }

  // --- Calendar ---
  const MONTHS = ['January','February','March','April','May','June',
                  'July','August','September','October','November','December'];

  function renderCalendar() {
    const y = viewYear, m = viewMonth;
    document.getElementById('calMonth').textContent = `${MONTHS[m]} ${y}`;

    const grid = document.getElementById('calGrid');
    grid.innerHTML = '';

    const firstDay = new Date(y, m, 1).getDay();
    const daysInMonth = new Date(y, m+1, 0).getDate();
    const tk = todayKey();

    // empty slots
    for (let i = 0; i < firstDay; i++) {
      const el = document.createElement('div');
      el.className = 'cal-day empty';
      grid.appendChild(el);
    }

    for (let d = 1; d <= daysInMonth; d++) {
      const key = toKey(y, m, d);
      const el = document.createElement('div');
      el.className = 'cal-day';
      el.textContent = d;
      if (key === tk) el.classList.add('today');
      if (data[key]) el.dataset.type = data[key];
      el.addEventListener('click', () => openModal(key, d));
      grid.appendChild(el);
    }
  }

  // --- Modal ---
  function openModal(key, day) {
    selectedDate = key;
    const [y, m, d] = key.split('-');
    document.getElementById('modalDate').textContent =
      `${MONTHS[parseInt(m)-1]} ${parseInt(d)}, ${y}`;
    document.getElementById('modalOverlay').classList.add('open');
  }

  function closeModal() {
    document.getElementById('modalOverlay').classList.remove('open');
    selectedDate = null;
  }

  function logActivity(type) {
    if (!selectedDate) return;
    data[selectedDate] = type;
    saveData();
    renderCalendar();
    renderHeatmap();
    renderStreak();
    closeModal();
  }

  // --- Heatmap ---
  function heatmapColor(type, count) {
    if (!type || count === 0) return '#1e1e22';
    if (type === 'lift') {
      if (count >= 4) return '#60a5fa';
      if (count === 3) return '#2563eb';
      if (count === 2) return '#1d4ed8';
      return '#1a3a80';
    }
    if (type === 'run') {
      if (count >= 4) return '#f87171';
      if (count === 3) return '#dc2626';
      if (count === 2) return '#b91c1c';
      return '#7a1212';
    }
    return '#2a2a2e'; // rest
  }

  // Count recent occurrences per type in a rolling window for intensity
  function getIntensity(key) {
    const type = data[key];
    if (!type || type === 'rest') return { type, count: type === 'rest' ? 1 : 0 };
    // Count how many of same type in past 7 days around this date
    let count = 0;
    for (let i = -3; i <= 3; i++) {
      const k = addDays(key, i);
      if (data[k] === type) count++;
    }
    return { type, count };
  }

  function renderHeatmap() {
    const grid = document.getElementById('heatmapGrid');
    grid.innerHTML = '';

    const today = new Date();
    today.setHours(0,0,0,0);

    for (let i = 89; i >= 0; i--) {
      const d = new Date(today);
      d.setDate(d.getDate() - i);
      const key = d.toISOString().slice(0,10);

      const { type, count } = getIntensity(key);
      const color = heatmapColor(type, count);

      const cell = document.createElement('div');
      cell.className = 'hm-cell';
      cell.style.background = color;

      // Tooltip via title
      const label = type ? type.charAt(0).toUpperCase() + type.slice(1) : 'No activity';
      cell.title = `${key} — ${label}`;
      cell.addEventListener('click', () => openModal(key, parseInt(key.split('-')[2])));

      grid.appendChild(cell);
    }
  }

  // --- Navigation ---
  document.getElementById('prevBtn').addEventListener('click', () => {
    viewMonth--;
    if (viewMonth < 0) { viewMonth = 11; viewYear--; }
    renderCalendar();
  });

  document.getElementById('nextBtn').addEventListener('click', () => {
    viewMonth++;
    if (viewMonth > 11) { viewMonth = 0; viewYear++; }
    renderCalendar();
  });

  // Close modal on overlay click
  document.getElementById('modalOverlay').addEventListener('click', (e) => {
    if (e.target === document.getElementById('modalOverlay')) closeModal();
  });

  // --- Init ---
  function init() {
    loadData();
    const now = new Date();
    viewYear = now.getFullYear();
    viewMonth = now.getMonth();
    renderCalendar();
    renderHeatmap();
    renderStreak();
  }

  init();
</script>
